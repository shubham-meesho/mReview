    public static ServiceRequest<AffiliateAttributesRequest> getAffiliateAttributesRequest(

    ) {
        if (!requestDetail.getDownStreamFlagsInfoRequestLevel().getAffiliateAttributesCallRequestMeta().isRequested()) {

    public static AffiliateAttributesDependencyData getAffiliateAttributesDependencyData(
            RequestDetail requestDetail,
            AffiliateAttributesResponse affiliateAttributesResponse
    ) {
        if (Objects.isNull(affiliateAttributesResponse)) {
            return null;
        }

        return AffiliateAttributesDependencyData.builder()
                .productIdToSupplierAffiliateDataMap(
                        requestDetail.getProductMetaInfoMap().values().stream()
                                .collect(Collectors.toMap(
                                        ProductMetaInfo::getProductId,
                                        productMetaInfo -> buildSupplierAffiliateData(productMetaInfo, affiliateAttributesResponse),
                                        (existing, replacement) -> replacement
                                ))
                )
                .build();
    }

    private static AffiliateAttributesDependencyData.SupplierAffiliateData buildSupplierAffiliateData(
            ProductMetaInfo productMetaInfo,
            AffiliateAttributesResponse response
    ) {
        Integer catalogId = productMetaInfo.getCatalogId();

        // Fetch catalog-level maps once, outside the per-supplier loop
        Map<Integer, Double> catalogCommissions = NullUtils.getNested(
                response, r -> r.getCatalogSuppliersCommissionPercentages().get(catalogId));
        Map<Integer, AffiliateInfo> catalogAffiliateInfos = NullUtils.getNested(
                response, r -> r.getCatalogSuppliersAffiliateInfo().get(catalogId));

        Map<Integer, Double> commissionPercentages = new HashMap<>();
        Map<Integer, AffiliateAttributesDependencyData.AffiliateInfo> affiliateInfoMap = new HashMap<>();

        // Single pass over suppliers — populates both maps at once
        for (SupplierMetaInfo supplierMetaInfo : productMetaInfo.getSupplierMetaInfoList()) {
            Integer supplierId = supplierMetaInfo.getSupplierId();

            if (catalogCommissions != null) {
                Double commission = catalogCommissions.get(supplierId);
                if (commission != null) {
                    commissionPercentages.put(supplierId, commission);
                }
            }

            if (catalogAffiliateInfos != null) {
                AffiliateInfo affiliateInfo = catalogAffiliateInfos.get(supplierId);
                if (affiliateInfo != null) {
                        affiliateInfoMap.put(supplierId,
                            AffiliateAttributesDependencyData.AffiliateInfo.builder()
                                    .reelzOffer(affiliateInfo.isReelzOffer())
                                    .reelzOfferText(affiliateInfo.getReelzOfferText())
                                    .build());
                }
            }
        }

        return AffiliateAttributesDependencyData.SupplierAffiliateData.builder()
                .commissionPercentages(commissionPercentages)
                .affiliateInfoMap(affiliateInfoMap)
                .build();
    }

    private static Map<Integer, Set<Integer>> getCatalogIdToSuppliersMap(
            RequestDetail requestDetail
    ) {
        try {
            return requestDetail.getProductMetaInfoMap().values().stream()
                    .filter(productMetaInfo -> !productMetaInfo.getSupplierMetaInfoList().isEmpty())
                    .collect(Collectors.toMap(
                            ProductMetaInfo::getCatalogId,
                            pmi -> pmi.getSupplierMetaInfoList().stream()
                                    .map(SupplierMetaInfo::getSupplierId)
                                    .collect(Collectors.toSet()),
                            (set1, set2) -> {
                                set1.addAll(set2);
                                return set1;
                            }
                    ));
        } catch (Exception e) {
            log.error("Error while getting catalogIdToSuppliers map : ", e);
            return null;
        }
    }
}
